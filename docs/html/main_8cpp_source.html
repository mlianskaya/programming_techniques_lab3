44template <typename T>
45double stddev(const std::vector<T>& sample) {
46 double m = mean(sample);
47 double sq = 0.0;
48 for (const auto& x : sample) sq += (x - m) * (x - m);
49 return std::sqrt(sq / sample.size());
50}

72template <typename T>
73std::vector<bool> toBits(const std::vector<T>& sample, int bitsPerNumber) {
74 std::vector<bool> bits;
75 bits.reserve(sample.size() * bitsPerNumber);
76 for (T x : sample) {
77 for (int b = 0; b < bitsPerNumber; ++b) {
78 bits.push_back((x >> b) & 1);
79 }
80 }
81 return bits;
82}

96generateSamples(Generator& gen, size_t numSamples, size_t sampleSize) {
97 std::vector<std::vector<typename Generator::result_type>> samples(numSamples);
98 for (size_t i = 0; i < numSamples; ++i) {
99 samples[i].reserve(sampleSize);
100 for (size_t j = 0; j < sampleSize; ++j) {
101 samples[i].push_back(gen.next());
102 }
103 }
104 return samples;

117template <typename T>
118void printStats(const std::string& name,
119 const std::vector<std::vector<T>>& samples,
120 T minVal, T maxVal,
121 double significance) {
122 std::cout << "\n=== " << name << " ===\n";
123 size_t passed = 0;
124 for (size_t i = 0; i < samples.size(); ++i) {
125 double m = mean(samples[i]);
126 double sd = stddev(samples[i]);
127 double cv = variability(samples[i]);
128 bool chi_ok = custom::chi_squared_test(samples[i],
129 minVal, maxVal,
130 significance,
131 custom::DistributionType::UNIFORM_INT);
132 if (chi_ok) ++passed;
133 std::cout << "Sample " << std::setw(2) << i+1
134 << ": mean=" << std::fixed << std::setprecision(2) << m
135 << ", std=" << sd
136 << ", var.coef=" << cv
137 << ", Chi-square=" << (chi_ok ? "PASS" : "FAIL") << "\n";
138 }
139 std::cout << "Chi-square passed: " << passed << "/" << samples.size() << "\n";
140}

151template <typename T>
152void runNISTTests(const std::string& name,
153 const std::vector<std::vector<T>>& samples,
154 double significance) {
155 std::cout << "\n--- NIST tests for " << name << " ---\n";
156 std::vector<bool> allBits;
157 for (const auto& sample : samples) {
158 auto bits = toBits(sample, sizeof(T) * 8);
159 allBits.insert(allBits.end(), bits.begin(), bits.end());
160 }
161
162 nist::FrequencyTest freq(allBits);
163 nist::BlockFrequencyTest blockFreq(allBits, 8);
164 nist::RunsTest runs(allBits);
165 nist::LongestRunTest longestRun(allBits);
166 nist::CumulativeSumsTest cumSums(allBits);
167
168 auto status = [&](double p) { return (p >= significance) ? "PASS" : "FAIL"; };
169 std::cout << "Frequency test: " << status(freq.p_value) << "\n";
170 std::cout << "Block frequency test: " << status(blockFreq.p_value) << "\n";
171 std::cout << "Runs test: " << status(runs.p_value) << "\n";
172 std::cout << "Longest run test: " << status(longestRun.p_value) << "\n";
173 std::cout << "Cumulative sums test: " << status(cumSums.p_value) << "\n";
174}

185template <typename Generator>
186std::vector<uint64_t> measureTimes(Generator& gen, const std::vector<size_t>& sizes) {
187 std::vector<uint64_t> times;
188 for (size_t n : sizes) {
189 auto start = std::chrono::high_resolution_clock::now();
190 volatile uint64_t tmp; // volatile предотвращает оптимизацию
191 for (size_t j = 0; j < n; ++j) tmp = gen.next();
192 auto end = std::chrono::high_resolution_clock::now();
193 times.push_back(std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count());
194 }
195 return times;
196}
