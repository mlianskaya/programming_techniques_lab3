38template <typename Number>
39std::pair<double, std::size_t> chi_squared(const std::vector<Number>& sample,
40 Number min, Number max,
41 DistributionType type) {
42 // Проверка корректности входных параметров
43 if (min >= max) {
44 std::ostringstream oss;
45 oss << "Range minimum " << min << " is not less than range maximum " << max;
46 throw std::invalid_argument(oss.str());
47 }
48 if (sample.size() < 2) {
49 std::ostringstream oss;
50 oss << "Sample size " << sample.size() << " is too small (need at least 2)";
51 throw std::invalid_argument(oss.str());
52 }
53
54 //Определяем длину интервала
55 int add = (type == DistributionType::UNIFORM_INT) ? 1 : 0;
56 double interval = static_cast<double>(max) - min + add;
57
58 // Количество интервалов (карманов) k = floor(log2(n)) + 1
59 std::size_t num_buckets = static_cast<size_t>(std::log2(sample.size()) + 1);
60 // Для целочисленных: если интервал меньше числа карманов, ограничиваем числом возможных значений
61 if (type == DistributionType::UNIFORM_INT && static_cast<double>(num_buckets) > interval)
62 num_buckets = static_cast<size_t>(interval);
63
64 double bucket_size = interval / num_buckets; // ширина одного интервала
65
66 // Вычисляем границы интервалов
67 std::vector<double> edges(num_buckets + 1);
68 edges[0] = min;
69 for (size_t i = 1; i <= num_buckets; ++i)
70 edges[i] = min + bucket_size * i;
71
72 // Теоретические вероятности для каждого интервала (равномерное распределение)
73 // = (ширина интервала)/interval
74 std::vector<double> p_unif(num_buckets);
75 for (size_t i = 0; i < num_buckets; ++i)
76 p_unif[i] = (edges[i+1] - edges[i]) / interval;
77
78 // Наблюдаемые частоты: подсчёт количества элементов в каждом интервале
79 // obs[i] — количество элементов выборки, попавших в i-й интервал
80 std::vector<size_t> obs(num_buckets, 0);
81 for (Number val : sample) {
82 // Ищем позицию val в массиве границ (первый элемент, строго больший val)
83 auto it = std::upper_bound(edges.begin(), edges.end(), val);
84 size_t pos = std::distance(edges.begin(), it) - 1;
85 // Защита от выхода за границы (если val точно равно правой границе последнего интервала)
86 if (pos >= num_buckets) pos = num_buckets - 1;
87 ++obs[pos];
88 }
89
90 // Вычисление статистики хи-квадрат
91 // χ² = Σ (obs_i - exp_i)^2 / exp_i
92 // Возвращаем χ² и степени свободы df = k - 1
93 double chi = 0.0;
94 for (size_t i = 0; i < num_buckets; ++i) {
95 double expected = p_unif[i] * sample.size();
96 double diff = obs[i] - expected;
97 chi += diff * diff / expected;
98 }
99 return {chi, num_buckets - 1};
100}

112template <typename Number>
113bool chi_squared_test(const std::vector<Number>& sample,
114 Number min, Number max, double significance,
115 DistributionType type) {
116 // Уровень значимости должен быть строго между 0 и 1
117 if (significance <= 0 || significance >= 1) {
118 std::ostringstream oss;
119 oss << "Significance level " << significance << " is not in (0; 1) interval";
120 throw std::invalid_argument(oss.str());
121 }
122
123 //Вычисляем χ² и df с помощью chi_squared
124 auto [chi, df] = chi_squared(sample, min, max, type);
125
126 // Для больших df используем нормальную аппроксимацию (центральная предельная теорема)
127 // χ² приблизительно нормально с параметрами df, 2df
128 // Возвращаем true, если p-value > significance (или если χ² меньше критического значения)
129 if (df > 30) {
130 // Нормированное отклонение: (χ² - df) / √(2df)
131 double z = (chi - df) / std::sqrt(2 * df);
132 // p-value = 1 - Φ(z) = 0.5 * erfc(z / √2)
133 // Ф — функция распределения стандартного нормального закона
134 double p = 0.5 * std::erfc(z / std::sqrt(2.0));
135 return p > significance;
136 } else {
137 // Для малых df используем заранее вычисленные критические значения для уровня 0.05
138 static const double critical[] = {3.84, 5.99, 7.81, 9.49, 11.07, 12.59, 14.07, 15.51, 16.92, 18.31};
139 if (df <= 10)
140 return chi <= critical[df - 1];
141 else
142 // Приближённая формула для df > 10 (до 30): χ² ≤ df + 2√(2df)
143 return chi <= df + 2 * std::sqrt(2 * df);
144 }
145}
