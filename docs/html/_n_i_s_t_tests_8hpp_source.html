71 BlockFrequencyTest(const std::vector<bool>& bits, size_t blockSize) {
72 size_t M = blockSize;
73 size_t N = bits.size() / M;
74 if (N == 0) { p_value = 0; return; }
75 double chi2 = 0.0;
76 // Цикл проходит по каждому блоку по порядку
77 for (size_t i = 0; i < N; ++i) {
78 size_t cnt = 0; // счётчик единиц в текущем блоке
79 // Внутренний цикл перебирает M битов текущего блока
80 for (size_t j = 0; j < M; ++j)
81 // Если бит равен 1 (истина), то cnt увеличивается
82 if (bits[i * M + j]) ++cnt;
83 double pi = static_cast<double>(cnt) / M; // выборочная доля единиц в текущем блоке
84 chi2 += (pi - 0.5) * (pi - 0.5);
85 }
86 chi2 *= 4.0 * M;
87 // Для больших N используем нормальную аппроксимацию
88 double mean = N;
89 double var = 2 * N;
90 double x = (chi2 - mean) / std::sqrt(var);
91 p_value = erfc(std::abs(x) / std::sqrt(2.0));
92 }
93 };

103 struct RunsTest {
104 double p_value;
105
106 RunsTest(const std::vector<bool>& bits) {
107 // Подсчитываем количество единиц во всей последовательности
108 int64_t ones = std::count(bits.begin(), bits.end(), true);
109 // Вычисляем выборочную долю единиц π
110 double pi = static_cast<double>(ones) / bits.size();
111
112 // Предварительная проверка: если π сильно отклоняется от 0.5, тест не проводится
113 // 2/√n – это эмпирическое правило, используемое в NIST
114 if (std::abs(pi - 0.5) > 2.0 / std::sqrt(bits.size())) {
115 p_value = 0.0;
116 return;
117 }
118
119 // Начинаем подсчёт серий. Первая серия всегда существует, поэтому runs начинается с 1
120 int64_t runs = 1;
121 // Проходим по последовательности, начиная со второго бита
122 for (size_t i = 1; i < bits.size(); ++i)
123 // Если текущий бит отличается от предыдущего, значит, началась новая серия – увеличиваем счётчик runs
124 if (bits[i] != bits[i-1]) ++runs;
125 // Вычисляем числитель статистики
126 double num = static_cast<double>(runs) - 2.0 * bits.size() * pi * (1.0 - pi);
127 //Знаменатель
128 double den = 2.0 * std::sqrt(2.0 * bits.size()) * pi * (1.0 - pi);
129 p_value = erfc(std::abs(num) / den);
130 }
131 };

142 struct LongestRunTest {
143 double p_value;
144
145 LongestRunTest(const std::vector<bool>& bits) {
146 size_t blockSize = 8;
147 size_t N = bits.size() / blockSize;
148 std::vector<int> maxRuns(N, 0); // вектор для хранения максимальной длины серии единиц (run) в каждом блоке
149
150 // Для каждого блока i перебираются биты
151 for (size_t i = 0; i < N; ++i) {
152 int cur = 0, best = 0;
153 for (size_t j = 0; j < blockSize; ++j) {
154 // Если бит равен 1, текущая длина cur увеличивается; если 0 – обнуляется
155 if (bits[i*blockSize + j]) cur++;
156 else cur = 0;
157 //В переменной best сохраняется максимальная длина серии единиц в этом блоке
158 if (cur > best) best = cur;
159 }
160 maxRuns[i] = best;
161 }
162 // Категории: 0–1, 2, 3, 4, 5, 6, ≥7
163 std::vector<int> obs(7,0);
164 for (int r : maxRuns) {
165 if (r <= 1) obs[0]++;
166 else if (r == 2) obs[1]++;
167 else if (r == 3) obs[2]++;
168 else if (r == 4) obs[3]++;
169 else if (r == 5) obs[4]++;
170 else if (r == 6) obs[5]++;
171 else obs[6]++;
172 }
173 // Теоретические вероятности для блоков длиной 8 бит (взяты из документа NIST SP 800‑22)
174 std::vector<double> exp = {0.2148, 0.3672, 0.2305, 0.0977, 0.0312, 0.0084, 0.0502};
175 double chi2 = 0.0;
176 for (size_t i = 0; i < 7; ++i) {
177 double e = exp[i] * N; // ожидаемое количество блоков в i‑й категории
178 // obs[i] – наблюдаемое
179 chi2 += (obs[i] - e) * (obs[i] - e) / e;
180 }
181 // Приближённое p-value
182 p_value = std::exp(-chi2 / 2);
183 }
184 };

195 struct CumulativeSumsTest {
196 double p_value;
197
198 CumulativeSumsTest(const std::vector<bool>& bits) {
199 int64_t S = 0; // текущая кумулятивная сумма
200 int64_t sup = 0, inf = 0;
201 // Цикл проходит по всем битам последовательности
202 for (bool b : bits) {
203 // если бит = 1, прибавляем +1; если 0, вычитаем 1
204 S += (b ? 1 : -1);
205 // Обновляем sup и inf, если текущее значение вышло за предыдущие границы
206 if (S > sup) sup = S;
207 if (S < inf) inf = S;
208 }
209 // Нормируем отклонения на квадратный корень из длины последовательности
210 double z1 = std::abs(sup) / std::sqrt(bits.size()); // нормированный положительный максимум
211 double z2 = std::abs(inf) / std::sqrt(bits.size()); // нормированный отрицательный минимум
212
213 // P(z) – это сумма по нечётным членам, дающая вероятность того, что максимум (по модулю) меньше z
214 auto P = [](double z) {
215 double sum = 0;
216 // Суммирование по k от -5 до 5 даёт достаточную точность, так как члены с большими |k| пренебрежимо малы
217 for (int k = -5; k <= 5; ++k) {
218 // разность дополнительных функций ошибок, которая эквивалентна разности нормальных CDF
219 double term = erfc((2*k+1)*z / std::sqrt(2.0)) - erfc((2*k-1)*z / std::sqrt(2.0));
220 sum += term;
221 }
222 return sum;
223 };
224 // итоговый p‑value – это вероятность того, что при случайном блуждании и положительное, и отрицательное отклонения будут не больше наблюдаемых
225 p_value = std::min(P(z1), P(z2));
226 }
227 };
